## Instalación

```bash
pnpm dlx shadcn@latest add @react-bits/ProfileCard-JS-CSS
```

## Uso

```tsx
import ProfileCard from "@/components/ProfileCard";
```

```tsx
<ProfileCard
  name="Juan Sánchez Vinuesa"
  title="Telematic Engineer"
  handle="Jsanchezv2201"
  status="Online"
  contactText="Contact Me"
  avatarUrl="/avatar.jpg"
  showUserInfo
  enableTilt={true}
  enableMobileTilt={false}
  behindGlowColor="rgba(125, 190, 255, 0.67)"
  behindGlowEnabled
  innerGradient="linear-gradient(145deg,#60496e8c 0%,#71C4FF44 100%)"
/>
```
